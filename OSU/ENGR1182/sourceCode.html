            //Test #1: Circular

            int power = cartDirection * 40 * powerAdjust();
            int counts(568);
            motorSpeed(4, power);
            int currentCount = 0;

            while (currentCount < counts) {
                currentCount = getBackwardCounts();
            }

            power = - cartDirection * 40 * powerAdjust();
            motorSpeed(4, 0);
            pauseFor(1.5);
            motorSpeed(4, 50);
            pauseFor(1.5);
            motorSpeed(4, 0);
        }
